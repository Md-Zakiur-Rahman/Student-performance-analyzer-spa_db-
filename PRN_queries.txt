-- Lists all subjects for admin and faculty controls.

-- Shows recent audit activity for the admin audit tab.
SELECT COALESCE(u.username, 'SYSTEM'), a.action, a.details, a.timestamp FROM audit_log a LEFT JOIN users u ON u.id = a.user_id ORDER BY a.timestamp DESC LIMIT %s;

-- Removes older view names so SHOW TABLES displays only the required view names.
DROP VIEW IF EXISTS vw_student_performance;

-- Removes older GPA view name before creating the required analytics view.
DROP VIEW IF EXISTS vw_student_gpa;

-- Removes older faculty access view name before creating the required views.
DROP VIEW IF EXISTS vw_faculty_subject_access;

-- Creates the student portal view used for student-facing marks display.
CREATE OR REPLACE VIEW v_student_portal AS SELECT s.id AS student_id, s.prn, s.name, s.department, sm.subject, sm.marks FROM students s LEFT JOIN student_marks sm ON sm.student_id = s.id;

-- Creates the analytics summary view used for dashboard totals and grade bands.
CREATE OR REPLACE VIEW v_analytics_summary AS SELECT COUNT(DISTINCT s.id) AS total_students, ROUND(AVG(sm.marks), 2) AS average_marks, MAX(sm.marks) AS highest_marks, MIN(sm.marks) AS lowest_marks, SUM(CASE WHEN sm.marks >= 90 THEN 1 ELSE 0 END) AS grade_a_count, SUM(CASE WHEN sm.marks >= 75 AND sm.marks < 90 THEN 1 ELSE 0 END) AS grade_b_count, SUM(CASE WHEN sm.marks >= 60 AND sm.marks < 75 THEN 1 ELSE 0 END) AS grade_c_count, SUM(CASE WHEN sm.marks >= 40 AND sm.marks < 60 THEN 1 ELSE 0 END) AS grade_d_count, SUM(CASE WHEN sm.marks < 40 THEN 1 ELSE 0 END) AS grade_f_count FROM students s LEFT JOIN student_marks sm ON sm.student_id = s.id;

-- Creates the recent audit view used to review login and data-changing activity.
CREATE OR REPLACE VIEW v_audit_recent AS SELECT a.id, COALESCE(u.username, 'SYSTEM') AS username, a.action, a.details, a.timestamp FROM audit_log a LEFT JOIN users u ON u.id = a.user_id ORDER BY a.timestamp DESC;
